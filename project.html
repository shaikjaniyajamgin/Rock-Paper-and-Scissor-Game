from flask import Flask, request import random app = Flask(__name__) @app.route('/', methods=['GET', 'POST']) def game(): html_template = """ Rock Paper Scissors

Rock, Paper, Scissors

{% if result %}

You chose: {{ user }}

Computer chose: {{ comp }}

{{ result }}

{% endif %}
""" result = '' user_choice = '' comp_choice = '' if request.method == 'POST': choices = ['rock', 'paper', 'scissors'] user_choice = request.form['choice'] comp_choice = random.choice(choices) if user_choice == comp_choice: result = "It's a tie!" elif (user_choice == 'rock' and comp_choice == 'scissors') or \ (user_choice == 'scissors' and comp_choice == 'paper') or \ (user_choice == 'paper' and comp_choice == 'rock'): result = "You win!" else: result = "You lose!" from jinja2 import Template template = Template(html_template) return template.render(result=result, user=user_choice, comp=comp_choice) if __name__ == '__main__': app.run(debug=True)
